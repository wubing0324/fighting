地图弹窗

```javascript
let overlayObj = null;
let _mapInstance = null;
export const addOverlay = (
  mapInstance: any,
  element: Element,
  lng: number,
  lat: number,
  offset: number[]
) => {
  if (overlayObj && overlayObj.remove && _mapInstance === mapInstance) {
    // 不同弹窗的位置偏移量不同，需要销毁重建了
    overlayObj.remove();
  }
  _mapInstance = mapInstance;
  overlayObj = mapInstance.createOverlay({ element, offset }, [lng, lat]);
  return overlayObj;
};

export const closeOverlay = () => {
  if (!overlayObj) return;
  overlayObj.remove();
};
```

```javascript
export const popupMapDialog = (
  el: Element,
  point: PointInfo,
  offset: number[] = [10, -70]
) => {
  const map = store.getMapInstance();
  zoom = point.zoom;
  popupPointId.value = point.id;
  addOverlay(map.instance, el, point.lng, point.lat, offset);
};

export const closeMapDialog = () => {
  closeOverlay();
  store.layersInfo.current = null;
  if (!zoom) return;
  store.getMapInstance().instance.map.setZoom(zoom);
};
```

```javascript
const pointClick = (point: PointInfo) => {
  if (store.layersInfo.current?.id === point.id) {
    closeDialog();
    return;
  }
  store.layersInfo.current = point;
  popupMapDialog(toElement(DialogStop), point);
};

const closeDialog = () => {
  closeMapDialog();
  store.layersInfo.current = null;
};
```
